出入口：

```xml
<include>
    <gateway name="IPLink">
      <param name="username" value="提供的电话号码"/>
      <param name="password" value="提供的密码"/>
      <param name="from-domain" value="sipit.iplink.no"/>
      <param name="realm" value="sipit.iplink.no"/>
      <param name="proxy" value="sipit.iplink.no"/>
      <param name="register-proxy" value="sipit.iplink.no"/>
      <param name="context" value="public"/>
      <param name="expire-seconds" value="1800"/>
      <param name="register" value="true"/>
      <param name="retry-seconds" value="60"/>
    </gateway>
</include> 
```
